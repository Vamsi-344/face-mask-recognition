def convert_img_to_pix(images):
  x=[]
  for i in range(len(images)):
    images[i]=cv2.resize(images[i],(64,64))
    a=[]
    for j in range(len(images[i])):
      for k in range(len(images[i][j])):
        for l in images[i][j][k]:
          a.append(l)
    x.append(a)
  return x
